// Callback hell — cada nivel añade complejidad loginUser(credentials, function(user) { loadProfile(user.id, function(profile) { loadPermissions(profile.role, function(permissions) { loadPreferences(user.id, function(prefs) { renderDashboard(profile, permissions, prefs); // 4 niveles. Añade gestión de errores y explota. }, onError); }, onError); }, onError); }, onError);
// Promises — sin pirámide, pero el flujo no es obvio loginUser(credentials) .then(user => loadProfile(user.id)) .then(profile => loadPermissions(profile.role)) .then(permissions => loadPreferences(permissions)) .then(prefs => renderDashboard(prefs)) .catch(error => showError(error.message)) .finally(() => hideLoader());
// Async/Await — parece síncrono, es asíncrono async function initDashboard(credentials) { try { const user = await loginUser(credentials); const profile = await loadProfile(user.id); const permissions = await loadPermissions(profile.role); const prefs = await loadPreferences(permissions); renderDashboard(profile, permissions, prefs); } catch(error) { showError(error.message); // un solo catch para todo el flujo } finally { hideLoader(); // siempre se ejecuta } }
